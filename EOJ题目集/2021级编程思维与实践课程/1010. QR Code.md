void itob(unsigned n, char s[])
{ 
	int i;
	for(i = 98; n && i; n /= 2){
		s[i--] = (((n % 2) < 10) ? '0' : ('A' - 10)) + (n % 2);
	}
	memmove(s , s + i + 1, 99 - i- 1);
	s[99 - i- 1] = '\0';
}

void PrintBit(char s[], int length)
{
	int len = strlen(s);
	if(len < length){
		for(int i = 0; i < (length - len); i ++){
			printf("0");
		}
	}
	printf("%s", s);
}
