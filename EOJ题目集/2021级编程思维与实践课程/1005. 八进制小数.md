void FloadDivide(char in[], char out[])
{
	int len = strlen(in);
	int temp, remainder = in[0] - '0', quotient = 0, index = 0;
	for(int i = 0 ; i < len || remainder != 0; i ++){
		if(i >= len - 1){
			temp = remainder * 10;

	}
}
