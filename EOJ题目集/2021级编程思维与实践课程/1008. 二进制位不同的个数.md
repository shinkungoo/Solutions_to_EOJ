void itob(int n, char s[])
{
    int i;
    for(i = 98; n && i; n /= 2){
        s[i--] = (((n % 2) < 10) ? '0' : ('A' - 10)) + (n % 2);
    }
    memmove(s , s + i + 1, 99 - i- 1);
    s[99 - i- 1] = '\0';
}

int compare(char s1[], char s2[])
{
	int diff = strlen(s1) - strlen(s2);
	int ret = 0;
	for(int i = 0; i < diff; i ++){
		if(s1[i] != '0'){
			ret ++;
		}
	} 
	for(int i = 0; s2[i]; i ++){
		if(s1[i + diff] != s2[i]){
			ret ++;
		}
	}
	
	return ret;
}
