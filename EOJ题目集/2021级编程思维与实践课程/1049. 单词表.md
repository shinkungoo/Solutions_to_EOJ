void qsort(char words[][20], int left, int right)
{
    int i, last;
    if(left >= right){
        return ;
    }
    swap(words, left, (left + right) / 2);
    last = left;
    for(i = left + 1; i <= right; i ++){
        if(strcmp(words[i], words[left]) < 0){
            swap(words, i, ++last);
        }
    }
    swap(words, left, last);
    qsort(words, left, last - 1);
    qsort(words, last + 1, right);
}
