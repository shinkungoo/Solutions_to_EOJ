void IntLayout(int num)
{
	int c = sizeof(num);
	unsigned char *p = (unsigned char *) &num;
	// get the address of each bit
	while(c --){
		printf("%02x%c", *(p ++), c == 0 ? '\n': ' ');
	}
}

void DoubleLayout(double num)
{
	int c = sizeof(num);
	unsigned char *p = (unsigned char *) &num;
	while(c --){
		printf("%02x%c", *(p ++), c == 0 ? '\n': ' ');
	}
}

	}
